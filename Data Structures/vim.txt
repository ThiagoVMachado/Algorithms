#include <bits/stdc++.h>
using namespace std;


double L, R, master;
double quero,topo;
double comeco(){
	double pai = L, mid;
	for(int i = 0 ; i < 100 ; i++){
		mid = (L + R)/2.0; 
		double area = (pai -mid)*(pai - mid) / (2.0*pai);
		if(area < quero) L = mid;
		else R = mid;
	}
	
	topo = (pai-mid)/pai;
	L = mid;
	R = 0;
	return master-L; 
}
double resto(){
	double pai = L, mid, h;
	for(int i = 0 ; i < 100 ; i++){
		mid = (L+R)/2.0;
		h = pai - mid;
		double area = ((h/pai) + (double)topo)/2.0*h;	
		if(area < quero) L = mid;
		else R = mid;
	}
	L = mid;
	R = 0;
	topo = (pai-mid)/h;
	return master-L;
}

int main(){
	double h;
	int pieces;
	cin >> pieces >> h;
	master = h;
	if(pieces == 1){
		cout << h << endl;
		return 0;
	}
	L = h;
	R = 0;
	quero = (h/2.0)/pieces;
	printf("%.10lf ", comeco());
	cout << "sobrou o trapÃ©zio de altura " << L << endl;
	for(int i = 2 ; i < pieces; i++){			
		printf("%.10lf ", resto());
	}
	cout << endl;
}
