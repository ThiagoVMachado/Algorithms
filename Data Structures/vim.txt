#include <bits/stdc++.h>
using namespace std;

double L, R, master;
double quero,base_menor;
double comeco(){
	double pai = L, mid;
	for(int i = 0 ; i < 100 ; ++i){
		mid = (L + R)/(double)2.0; 
		double area = (pai -mid)*(pai - mid) /(double)((double)2.0*pai);
		if(area < quero) L = mid;
		else R = mid;
	}
	base_menor = 1.0 - (mid/pai) ;
	cout << "topo vale " << base_menor << endl;
	L = mid;
	R = 0.0;
	return master-L; 
}
double resto(){
	double pai = L, mid, h, base_maior;
	for(int i = 0 ; i < 100 ; ++i){
		mid = (L+R)/2.0;
		base_maior = 1.0 - (mid/pai);
		h = pai - mid;
		double area = (base_maior + base_menor)*h/2.0;	
		if(area < quero) L = mid;
		else R = mid;
	}
	L = mid;
	R = 0;
	base_menor = base_maior;
	cout << "topo vale " << base_menor << endl;
	return master-L;
}
